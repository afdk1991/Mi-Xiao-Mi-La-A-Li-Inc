import datetime
import os

# 全域刚性固定配置（禁止私自修改）
PROJECT_CODE = "MIXMLAAL"       # 全域唯一项目标识符
MAJOR = 0                       # 主版本固化基线
MINOR = 0                       # 次版本固化基线
REVISION = 0                    # 修订版本固化基线
VERSION_FILE = "version_record.txt"  # 全域版本留痕台账文件

def get_real_time():
    """同步系统全域实时时间戳，严格输出YYYYMMDDHHMMSS标准格式"""
    return datetime.datetime.now().strftime("%Y%m%d%H%M%S")

    """自动读取历史构建版本，有序递增，首次运行默认接续基线4"""

        current_build = 4

    return full_version, build
